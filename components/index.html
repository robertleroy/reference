# Accordion

      <script>
    export default {
      name: 'Accordion',
      props: {
        opened: {
          type: [Boolean, String],
          default: false,
          required: false
        },
      },
      data() {
        return {
          isOpen: this.opened,
        };
      },
      methods: {
        start(el) {
          el.style.height = el.scrollHeight + 'px';
        },
        end(el) {
          el.style.height = '';
        }
      }
    };
  </script>

  <template>
    <section class='accordion'>
      <div class="header" @click="isOpen = !isOpen">
        <slot name="header"></slot>
      </div>
      
      <transition 
        name="collapse"
        @enter="start"
        @after-enter="end"
        @before-leave="start"
        @after-leave="end">
        <div class="content" v-show="isOpen">
          <div class="panel">
            <slot name="panel"></slot>
          </div>
        </div>
      </transition>
    </section>
  </template>
