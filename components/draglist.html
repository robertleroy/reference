# Vue Draggable

Documentation (opens new window)
Examples (opens new window)

<!-- npm i -S vuedraggable@next -->

<script>
  import draggable from 'vuedraggable'
  export default {
    name: 'example',
    components: {     
      'draggable': draggable, 
    },
    data() {
      return {
        drag: false,
        numbers: [
          { name: 'One', id: 1 },
          { name: 'Two', id: 2 },
          { name: 'Three', id: 3 }
        ],
        letters: [
          { name: 'A', id: 'a' },
          { name: 'B', id: 'b' },
          { name: 'C', id: 'c' }
        ],        
      };
    },
    methods: { 
      clone: function(el) {
        return { name: el.name + " cloned" };
      },
    }
  };
</script>

<template>
  <section id='example'>
      <div class="row">
        <div class="col">
          <draggable 
            class="list-group"
            :list="numbers"
            group="lemons"
            itemKey="id">
            <template #item="{ element, index }">
              <div class="list-group-item">{{ element.name }} {{ index }}</div>
            </template>
          </draggable>
        </div>

        <div class="col">
          <draggable 
            class="list-group"
            :list="letters"
            group="lemons"
            itemKey="id">
            <template #item="{ element, index }">
              <div class="list-group-item">{{ element.name }} {{ index }}</div>
            </template>
          </draggable>
        </div> <!-- col -->
      </div> <!-- row --> 
  </section>
</template>
